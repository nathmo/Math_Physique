\section*{Chapter 6 -- Energy}

\rappelbox{green}{%
\begin{itemize}
  \item $E_{p} = mgh$ \quad (gravitational potential energy)
  \item $E_{c} = \dfrac{1}{2}mv^{2}$ \quad (kinetic energy)
  \item $E_{th} = mc\,\Delta T$ \quad (thermal energy, $c = 4180\,\text{J/(kg\ensuremath{\cdot}K)}$ for water)
  \item $E_{el} = UIt$ \quad (electrical energy)
  \item \textbf{Conservation:} $E_{\text{input}} = E_{\text{output}} + E_{\text{lost}}$
\end{itemize}
}

\section*{Chapter 7 -- Power and work}

\rappelbox{orange}{%
\begin{itemize}
  \item $P = \dfrac{E}{t}$ \quad (power = energy / time)
  \item $P = Fv$ \quad (mechanical power)
  \item $W = F \times d \times \cos(\theta)$ \quad (work done by a force)
  \item $1\,\text{W} = 1\,\text{J/s}$
  \item $1\,\text{kWh} = 3.6 \times 10^{6}\,\text{J}$
\end{itemize}
}
